# lecture/criture d'un fichier Excel avec xlwings
def save_with_xlwings(file):
    tempfile = file
    excel_app = xw.App(visible=False)
    excel_book = excel_app.books.open(file)
    excel_book.save(tempfile)
    excel_book.close()
    excel_app.quit()
    return tempfile
